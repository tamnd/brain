$$3+7+3+7 = 20$$

tens = {

    40: len("forty"),
    50: len("fifty"),
    60: len("sixty"),
    70: len("seventy"),
    80: len("eighty"),
    90: len("ninety"),
}

def letters(n):
    if n == 1000:
        return len("one") + len("thousand")

    if n >= 100:
        h = ones[n // 100] + len("hundred")
        if n % 100 == 0:
            return h
        return h + len("and") + letters(n % 100)

    if n >= 20:
        return tens[(n // 10) * 10] + letters(n % 10)

    if n == 0:
        return 0

    return ones[n]

total = sum(letters(i) for i in range(1, 1001))
print(total)
