[Project Euler Problem 896](https://projecteuler.net/problem=896)

## Solution

**Answer:** `274229635640`

A range $[a..a+L-1]$ is divisible iff we can assign each divisor $i \in \{1,\dots,L\}$ to a distinct offset $j\in\{0,\dots,L-1\}$ such that

$$a+j \equiv 0 \pmod i$$

Equivalently,

$$a \equiv -j \pmod i$$

So every valid permutation corresponds to a consistent system of congruences.

We can enumerate all such congruence systems using backtracking + CRT (Chinese Remainder Theorem), then sort the resulting valid starting values.

Here is a complete Python solution:

```
#!/usr/bin/env python3

import math
from typing import List, Optional, Set, Tuple

def egcd(a: int, b: int) -> Tuple[int, int, int]:
    """Extended Euclidean algorithm."""
    x0, y0, x1, y1 = 1, 0, 0, 1
    while b:
        q = a // b
        a, b = b, a % b
        x0, x1 = x1, x0 - q * x1
        y0, y1 = y1, y0 - q * y1
    return a, x0, y0

def crt_merge(r1: int, m1: int,
              r2: int, m2: int) -> Optional[Tuple[int, int]]:
    """
    Merge:
        x ≡ r1 (mod m1)
        x ≡ r2 (mod m2)

    Returns:
        (r, lcm(m1,m2))
    or None if inconsistent.
    """
    g = math.gcd(m1, m2)

    if (r2 - r1) % g != 0:
        return None

    l = (m1 // g) * m2

    m1g = m1 // g
    m2g = m2 // g
    diff = (r2 - r1) // g

    _, inv, _ = egcd(m1g, m2g)
    inv %= m2g

    t = (diff * inv) % m2g

    r = (r1 + m1 * t) % l

    return r, l

def lcm_upto(n: int) -> int:
    """LCM(1,2,...,n)."""
    m = 1
    for i in range(1, n + 1):
        m = (m // math.gcd(m, i)) * i
    return m

def candidate_offsets(L: int,
                      unused_mask: int,
                      target: int,
                      step: int) -> List[int]:
    """
    All offsets j such that:
        j ≡ target (mod step)
    and j is still unused.
    """
    res = []

    j = target
    while j < L:
        if (unused_mask >> j) & 1:
            res.append(j)
        j += step

    return res

def pick_index_mrv(L: int,
                   r: int,
                   m: int,
                   unused_mask: int,
                   remaining_mask: int):
    """
    Minimum Remaining Values heuristic:
    choose the divisor i with the fewest feasible offsets.
    """
    best_i = 0
    best_cands = None
    best_count = 10**9

    for i in range(L, 0, -1):

        if ((remaining_mask >> (i - 1)) & 1) == 0:
            continue

        g = math.gcd(m, i)

        target = (-r) % g

        cands = candidate_offsets(L, unused_mask, target, g)

        c = len(cands)

        if c == 0:
            return None

        if c < best_count:
            best_count = c
            best_i = i
            best_cands = cands

    return best_i, best_cands

def enumerate_valid_residues(L: int):
    """
    Enumerate all valid residues a modulo M,
    where M = lcm(1..L).
    """
    M = lcm_upto(L)

    all_offsets_mask = (1 << L) - 1
    all_indices_mask = (1 << L) - 1

    residues: Set[int] = set()
    visited = set()

    def dfs(r: int,
            m: int,
            unused_mask: int,
            remaining_mask: int):

        r %= m

        key = (r, m, unused_mask, remaining_mask)

        if key in visited:
            return

        visited.add(key)

        if remaining_mask == 0:
            residues.add(r)
            return

        pick = pick_index_mrv(
            L, r, m, unused_mask, remaining_mask
        )

        if pick is None:
            return

        i, cands = pick

        remaining2 = remaining_mask & ~(1 << (i - 1))

        for j in cands:

            merged = crt_merge(
                r, m,
                (-j) % i, i
            )

            if merged is None:
                continue

            r2, m2 = merged

            dfs(
                r2,
                m2,
                unused_mask & ~(1 << j),
                remaining2
            )

    dfs(0, 1, all_offsets_mask, all_indices_mask)

    return residues, M

def nth_divisible_range_start(L: int, n: int) -> int:

    residues, M = enumerate_valid_residues(L)

    starts = []

    for r in residues:
        starts.append(r if r > 0 else M)

    starts.sort()

    return starts[n - 1]

# Small verification from the statement:
assert nth_divisible_range_start(4, 1) == 1
assert nth_divisible_range_start(4, 2) == 2
assert nth_divisible_range_start(4, 3) == 3
assert nth_divisible_range_start(4, 4) == 6

# Final computation:
print(nth_divisible_range_start(36, 36))
```

### Explanation of the approach

- Each number in the range is represented by an offset $j$.
- Assigning divisor $i$ to offset $j$ forces:

$$a+j \equiv 0 \pmod i$$

hence

$$a \equiv -j \pmod i$$
- We build these congruences incrementally.
- Whenever a new congruence is added, we merge it using the Chinese Remainder Theorem.
- Backtracking explores all consistent assignments.
- Finally we sort all valid starting values and select the 36th.

The small test from the problem statement is verified:

- length $4$:

$$[1..4], [2..5], [3..6], [6..9]$$

are indeed the first four divisible ranges.
