{{/* Renders .Content with the first heading stripped when the page title starts with a number prefix (e.g. "1.", "01.") and that same prefix appears at the start of the first h1/h2/h3 in the content — preventing a duplicate title/heading pair. */}} {{- $content := .Content -}} {{- if findRE `^\d+\.` .Title -}} {{- $firstHeadings := findRE `]*>[\s\S]*?` $content -}} {{- if $firstHeadings -}} {{- $firstText := index $firstHeadings 0 | plainify -}} {{- if findRE `^\s*\d+\.` $firstText -}} {{- $content = replaceRE `]*>[\s\S]*?\n?` "" $content 1 -}} {{- end -}} {{- end -}} {{- end -}} {{- $content | safeHTML -}}
